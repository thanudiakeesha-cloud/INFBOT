            let currentLang = localStorage.getItem('infinityLang') || 'en';

            // Apply translations
            function applyLanguage(lang) {
                document.querySelectorAll('[data-i18n]').forEach(el => {
                    const key = el.getAttribute('data-i18n');
                    if (translations[lang] && translations[lang][key]) {
                        // If the element has HTML inside (like <b>), we need to set innerHTML
                        // But most are plain text, we'll use innerHTML to preserve any existing inner HTML (like <span>)
                        // However, we must be careful not to break existing inner elements.
                        // A safe approach: if the original translation contains HTML, use innerHTML; else textContent.
                        const translation = translations[lang][key];
                        if (translation.includes('<') || translation.includes('&')) {
                            el.innerHTML = translation;
                        } else {
                            el.textContent = translation;
                        }
                    } else if (lang === 'si' && !translations.si[key]) {
                        // Fallback to English if Sinhala key missing
                        if (translations.en[key]) {
                            el.textContent = translations.en[key];
                        }
                    }
                });
                currentLang = lang;
                localStorage.setItem('infinityLang', lang);
            }

            // Apply stored language on load
            applyLanguage(currentLang);
